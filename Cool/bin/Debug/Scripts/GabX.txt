local function animate()
	button:TweenSizeAndPosition(UDim2.new(0, 300, 0, 40), UDim2.new(0.5, -150, 0.5, -20))
	wait(2)
	button.TextScaled = false
	button.Text = ""
	button:TweenSizeAndPosition(UDim2.new(0, 110, 0, 45), UDim2.new(0, 0, 0.7, 0))
	spawn(function()
		for i=20, 0, -1 do
			topthing.Size = UDim2.new(1, 0, 0, i)
			topthing.Position = UDim2.new(0, 0, 0, -i)
			bottomthing.Size = UDim2.new(1, 0, 0, i)
			wait()
		end
	end)
	spawn(function()
		for i=163, 132, -1 do
			button.BackgroundColor3 = Color3.fromRGB(i, i, i)
			wait()
		end
	end)
	spawn(function()
		wait(1)
		button.TextSize = 30
		button.Text = "OPEN"
		button.MouseButton1Click:connect(function()
			if top.Visible then
				button.Text = "OPEN"
				top:TweenPosition(UDim2.new(0, -505, 0.5, -218))
				wait(1.05)
				top.Visible = false
			else
				button.Text = "CLOSE"
				top.Visible = true
				top:TweenPosition(UDim2.new(0.5, -250, 0.5, -218))
				wait(1.05)
			end 
		end)
	end)
	wait(25/30)
	button.Style = Enum.ButtonStyle.RobloxRoundButton
end

	text.Size = UDim2.new(0, 390, 1, -10)
	text.Position = UDim2.new(0, 105, 0, 5)
	text.Text = tab.text
	text.Font = Enum.Font.SciFi
	text.TextSize = 20
	text.TextXAlignment = Enum.TextXAlignment.Center
	text.TextYAlignment = Enum.TextYAlignment.Top
	text.TextWrapped = true

	end

	end)
